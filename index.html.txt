    <div class="chat-container">
        <div class="chat-box" id="chat-box">
            <p class="bot-message">Hello! How can I assist you today?</p>
        </div>
        <div class="input-area">
            <input type="text" id="user-input" placeholder="Type your message here...">
            <button onclick="sendMessage()">Send</button>
        </div>
    </div>

    <script>
        const chatBox = document.getElementById('chat-box');
        const userInput = document.getElementById('user-input');

        // Function to append messages to chat
        function appendMessage(message, className) {
            const messageElement = document.createElement('p');
            messageElement.className = className;
            messageElement.innerText = message;
            chatBox.appendChild(messageElement);
            chatBox.scrollTop = chatBox.scrollHeight;
        }

        // Function to send a message
        function sendMessage() {
            const message = userInput.value;
            if (message.trim() === '') return;
            
            // Append user's message
            appendMessage(message, 'user-message');
            userInput.value = '';

            // Fetch response from OpenAI API
            fetch('https://api.openai.com/v1/chat/completions', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': 'Bearer YOUR_OPENAI_API_KEY'  // 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
                },
                body: JSON.stringify({
                    model: "gpt-3.5-turbo",
                    messages: [{ role: "user", content: message }]
                })
            })
            .then(response => response.json())
            .then(data => {
                const botReply = data.choices[0].message.content;
                appendMessage(botReply, 'bot-message');
            })
            .catch(error => {
                console.error('Error:', error);
                appendMessage('Sorry, something went wrong. Please try again later.', 'bot-message');
            });
        }

        // Send message on 'Enter' key press
        userInput.addEventListener('keypress', function (e) {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });
    </script>
